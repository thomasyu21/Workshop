Inside:
<VirtualHost *:80>
	ServerName <IP Address>
	ServerAdmin user@ip_address
	WSGIScriptAlias / /var/www/09_flask_v0/09_flask_v0.wsgi
	<Directory /var/www/09_flask_v0/>
		Order allow,deny
		Allow from all
	</Directory>
	ErrorLog ${APACHE_LOG_DIR}/error.log
	LogLevel warn
	CustomLog ${APACHE_LOG_DIR}/access.log combined
</VirtualHost>
